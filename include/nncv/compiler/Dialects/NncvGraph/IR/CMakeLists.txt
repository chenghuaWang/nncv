# add dialect
function(nncv_graph_add_dialect)
    set(LLVM_TARGET_DEFINITIONS NncvGraphOps.td)
    mlir_tablegen(NncvGraphAttrs.h.inc -gen-attrdef-decls)
    mlir_tablegen(NncvGraphAttrs.cpp.inc -gen-attrdef-defs)
    mlir_tablegen(NncvGraphEnums.h.inc -gen-enum-decls)
    mlir_tablegen(NncvGraphEnums.cpp.inc -gen-enum-defs)
    mlir_tablegen(NncvGraphOps.h.inc -gen-op-decls)
    mlir_tablegen(NncvGraphOps.cpp.inc -gen-op-defs)
    mlir_tablegen(NncvGraphTypes.h.inc -gen-typedef-decls)
    mlir_tablegen(NncvGraphTypes.cpp.inc -gen-typedef-defs)
    mlir_tablegen(NncvGraphDialect.h.inc --gen-dialect-decls --dialect=nncvgraph)
    mlir_tablegen(NncvGraphDialect.cpp.inc --gen-dialect-defs --dialect=nncvgraph)
    mlir_tablegen(NncvGraphAttributes.h.inc -gen-attrdef-decls)
    mlir_tablegen(NncvGraphAttributes.cpp.inc -gen-attrdef-defs)
    add_public_tablegen_target(MLIRNncvGraphIncGen)
    add_dependencies(mlir-headers MLIRNncvGraphIncGen)
endfunction(nncv_graph_add_dialect)

# add interface
function(nncv_graph_add_interface)
    set(LLVM_TARGET_DEFINITIONS NncvGraphInterface.td)
    mlir_tablegen(NncvGraphOpInterfaces.h.inc -gen-op-interface-decls)
    mlir_tablegen(NncvGraphOpInterfaces.cpp.inc -gen-op-interface-defs)
    mlir_tablegen(NncvGraphTypeInterfaces.h.inc -gen-type-interface-decls)
    mlir_tablegen(NncvGraphTypeInterfaces.cpp.inc -gen-type-interface-defs)
    add_public_tablegen_target(NncvGraphInterfacesIncGen)
    add_dependencies(MLIRNncvGraphIncGen NncvGraphInterfacesIncGen)
endfunction(nncv_graph_add_interface)

nncv_graph_add_dialect()
nncv_graph_add_interface()
